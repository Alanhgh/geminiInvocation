Gemini API调用示例代码（Python）


# Gemini API调用示例 - Python
import requests
import json

# 您的API密钥
API_KEY = "your_poloai_api_key"  
# API接口地址
API_URL = "https://poloai.top/v1/models/gemini-2.5-pro:generateContent"  

headers = {
    "Content-Type": "application/json",
    "x-goog-api-key": API_KEY
}

# 请求参数
data = {
    "contents": [
        {
            "parts": [
                {
                    "text": "请介绍Gemini API的主要特点和使用场景"
                }
            ]
        }
    ],
    "generationConfig": {
        "temperature": 0.7,
        "maxOutputTokens": 1000,
        "topP": 0.8
    }
}

# 发送请求
response = requests.post(API_URL, headers=headers, json=data)
print(json.dumps(response.json(), indent=4, ensure_ascii=False))

// Gemini API调用示例 - JavaScript (Node.js)
const axios = require('axios');

const API_KEY = 'your_poloai_api_key';
const API_URL = 'https://poloai.top/v1/models/gemini-2.5-pro:generateContent';

async function callGeminiAPI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        contents: [
          {
            parts: [
              {
                text: "请介绍Gemini API的主要特点和使用场景"
              }
            ]
          }
        ],
        generationConfig: {
          temperature: 0.7,
          maxOutputTokens: 1000,
          topP: 0.8
        }
      },
      {
        headers: {
          'Content-Type': 'application/json',
          'x-goog-api-key': API_KEY
        }
      }
    );
    
    console.log(JSON.stringify(response.data, null, 2));
  } catch (error) {
    console.error('调用Gemini API出错:', error.response?.data || error.message);
  }
}

callGeminiAPI();
